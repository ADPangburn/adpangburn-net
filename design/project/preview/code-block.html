A retry budget like retry_budget = 3 bottoms out in a human, not an exception:

orchestrator/retry.tstypescript
// retries bottom out in a human, not an exception
export async function retry(task: Task, budget = 3) {
  for (let attempt = 0; attempt < budget; attempt++) {
    try { return await task.run() }
    catch (e: ToolError) { logAttempt(e, attempt) }
  }
  return askHuman(task, { reason: "out of retries" })
}
